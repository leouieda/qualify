def adaptive_discretize(tesseroids, point, ratio):
    lon, lat, h = point
    d2r = np.pi/180.    
    radius = h + MEAN_EARTH_RADIUS
    mesh = []
    for tesseroid in tesseroids:
        size = max([MEAN_EARTH_RADIUS*d2r*(tesseroid.e - tesseroid.w),
                    MEAN_EARTH_RADIUS*d2r*(tesseroid.n - tesseroid.s),
                    tesseroid.top - tesseroid.bottom])
        tes_radius = tesseroid.top + MEAN_EARTH_RADIUS
        tes_lat = d2r*0.5*(tesseroid.s + tesseroid.n)
        tes_lon = d2r*0.5*(tesseroid.w + tesseroid.e)
        distance = np.sqrt(radius**2 + tes_radius**2 - 2.*radius*tes_radius*(
            np.sin(d2r*lat)*np.sin(tes_lat) + np.cos(d2r*lat)*np.cos(tes_lat)*np.cos(d2r*lon - tes_lon)))
        if distance < ratio*size:
            mesh.extend(adaptive_discretize(tesseroid.split(2, 2, 2), [lon, lat, h], ratio))
        else:
            mesh.append(tesseroid)
    return mesh
